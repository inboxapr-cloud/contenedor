steps
  # Instala las dependencias del proyecto
  - name 'node20'
    entrypoint npm
    args ['install']
  # Compila la aplicación de Next.js para producción
  - name 'node20'
    entrypoint npm
    args ['run', 'build']

      - '--source=.'
